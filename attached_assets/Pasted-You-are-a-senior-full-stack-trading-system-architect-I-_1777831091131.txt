// UT Bot trailing stop calculation
calcTrail(float prevTrail, float prevClose, float curClose, float nLoss) =>
    float result = curClose - nLoss
    if na(prevTrail)
        result := curClose - nLoss
    else if curClose > prevTrail and prevClose > prevTrail
        result := math.max(prevTrail, curClose - nLoss)
    else if curClose < prevTrail and prevClose < prevTrail
        result := math.min(prevTrail, curClose + nLoss)
    else if curClose > prevTrail
        result := curClose - nLoss
    else
        result := curClose + nLoss
    result
